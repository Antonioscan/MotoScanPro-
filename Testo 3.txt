function analizzaFoto() {
  const risultato = document.getElementById("risultato");
  const upload = document.getElementById("upload");

  if (!upload.files.length) {
    risultato.textContent = "Per favore, carica una foto.";
    return;
  }

  const nomeFile = upload.files[0].name.toLowerCase();

  // Simulazione riconoscimento componenti
  if (nomeFile.includes("filtro")) {
    risultato.textContent = "Rilevato: Filtro aria per Yamaha MT-07.";
  } else if (nomeFile.includes("leva")) {
    risultato.textContent = "Rilevato: Leva freno compatibile Ducati Panigale.";
  } else {
    risultato.textContent = "Ricambio non riconosciuto. Prova con un'altra foto.";
  }
}
